CREATE OR REPLACE VIEW elex_results as
    SELECT x.*, y.nyt_candidate_name,y.nyt_candidate_important,y.nyt_candidate_description,y.nyt_races,y.nyt_display_order,y.nyt_winner,y.nyt_delegates, z.* FROM results as z
        LEFT JOIN override_candidates as y on z.id = y.id
        LEFT JOIN override_races as x on z.raceid = x.race_raceid and z.statepostal = x.race_statepostal
;
